import java.io.InputStreamReader;
import java.io.IOException;

class C
{
   	public static void main(String st[]) throws IOException
	{
        		InputStreamReader obj = new InputStreamReader(System.in);
			System.out.println("Enter a character: ");
        		int str = obj.read();
   	     		System.out.println("Entered Character: " + (char)str);
	}
}

10. Get all characters from the file, count number of lines, words, characters and display on the screen

import java.util.Scanner;
import java.io.File;
import java.io.FileNotFoundException;

class str
{
    public static void main(String st[])
    {
        try
        {
            File file2= new File("file.txt");
            Scanner obj=new Scanner(file2);  
            int lines = 0;
            int words = 0;
            int characters = 0;
            while (obj.hasNextLine()) 
            {
                String line = obj.nextLine();
                lines++;
                words += line.split("\\s+").length;
                characters += line.length() + 1;//for new line
                System.out.println(line);
            }
            System.out.println("Lines: " + lines);
            System.out.println("Words: " + words);
            System.out.println("Characters: " + characters);
        }
        catch(java.io.IOException e)
        {
            System.out.println(e.getMessage());
        }
    }
}

Output

D:\javaprogram\input-output>javac input10.java

D:\javaprogram\input-output>java str
Aman
Lines: 1
Words: 1
Characters: 5
