5. Write a Java program to create a class called "Shape" with abstract methods for calculating area and perimeter, and subclasses for "Rectangle", "Circle", and "Triangle".

abstract class Shape
{
	 abstract double calculatingArea();
	 abstract double calculatingPerimeter();
}
class Rectangle extends Shape
{
	double width;
	double height;
	public Rectangle(int w , int h)
	{
		width=w;
		height=h;
	}
	public double calculatingArea()
	{
		return width*height;
	}
	public double calculatingPerimeter()
	{
		return 2*(width+height);
	}
}
class Circle extends Shape
{
	double radius;

	public Circle(double r)
	{
		radius=r;
	}
	public double calculatingArea()
	{
		return 3.14*radius*radius;
	}
	public double calculatingPerimeter()
	{
		return 2*3.14*radius;
	}
}
class Triangle extends Shape
{
	double base;
	double height;
	double side;
	
	public Triangle(int b ,int h,int s)
	{	
		base=b;
		height=h;
		side=s;
	}
	public double calculatingArea()
	{
		return height*base/2;	
	}
	public double calculatingPerimeter()
	{	
		return side+base+side;
	}
}
class Main
{
	public static void main(String args[])
	{
		Rectangle obj = new Rectangle(5,3);		
		System.out.println("Area of rectangle :"+ obj.calculatingArea());
		System.out.println("Perimeter of rectangle :"+ obj.calculatingPerimeter());
	
		Circle obj1 = new Circle(10);		
		System.out.println("Area of Circle :"+ obj1.calculatingArea());
		System.out.println("Perimeter of Circle :"+ obj1.calculatingPerimeter());
	
		Triangle obj2 = new Triangle(20,12,12);		
		System.out.println("Area of Traingle :"+ obj2.calculatingArea());
		System.out.println("Perimeter of Traingle :"+ obj2.calculatingPerimeter());
	}
}

Area of Circle :314.0

Area of Traingle :120.0
Perimeter of Traingle :44.0

8. Create a class showing an example of copy constructor.

class A
{
    int x;
    A(int y)
    {
        x=y;
    }
    A(A t)
    {
        x=t.x;
    }
    void display()
    {
        System.out.println(x);
    }
    public static void main(String st[])
    {
        A obj= new A(10);
        obj.display();
        obj.x++;
        A obj1=new A(obj);
        obj1.display();
    }
}
