1.Write a Java method to find the smallest number among three 

class A
{
	public static void main(String args[])
	{
		int a =10;
		int b =20;
		int c =30;	
	
		int smallest = findSmallest(a,b,c);
		System.out.println("Smallest number:"+smallest);
	}
	public static int findSmallest(int a,int b ,int c)
	{
		if(a<=b && a<=c)
		{
			return a;
		}
		else if(b<=a && b<=c)
		{
			return b;
		}
		else 
		{
			return c;
		}
	}
		

}

2.Write a Java method to compute the average of three numbers

class sum
{
	public static void main(String args[])
	{
		int a =10;
		int b =20;
		int c=30;
		
		int average=computeAverage(a,b,c);
		System.out.println("Average:"+average);
	}
	public static int computeAverage(int a ,int b ,int c)
	{
		return(a+b+c)/3;
	}
}

3.java method to count all word in a string 

class A
{
	public static void main(String args[])
	{
		String str="Aman";
		
		int wordCount=countWords(str);
		System.out.println("Word count:"+wordCount);
	}

		static int countWords(String str)
		{
			if(str==null)
			{
				return 0;
			}
			return str.split("").length;
		}
}

6. Write a Java method that checks whether all the characters in a given string are vowels (a, e,i,o,u) or not. Return true if each character in the string is a vowel, otherwise return false.

class CheckVowel
{
    public static void main(String st[])
    {
        System.out.println(IfVowel("abcd"));
        System.out.println(IfVowel("aeiou"));
        System.out.println(IfVowel("AEIOU"));         
    }  
        static boolean IfVowel(String str)
        {
            for(int i = 0; i < str.length(); i++)
            {
                char c = str.charAt(i);
                if((c!='a' && c!= 'e'&& c!='i' && c!= 'o' && c!='u' && c!='A' && c!= 'E'&& c!='I' && c!= 'O' && c!='U'))
                {
                  return false;
                }
            }
            return true;
        }
}
